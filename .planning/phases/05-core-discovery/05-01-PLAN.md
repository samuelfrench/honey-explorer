### 3. Create Honey API Service
**File**: `frontend/src/services/api.ts`

```typescript
import axios from 'axios';

const api = axios.create({
  baseURL: '/api',
});

export interface Honey {
  id: number;
  name: string;
  slug: string;
  description: string;
  floralSource: string;
  origin: string;
  honeyType: string;
  colorIntensity: string;
  flavorProfiles: string;
  texture: string;
  imageUrl: string;
  thumbnailUrl: string;
  featured: boolean;
}

export const honeyApi = {
  getAll: () => api.get<Honey[]>('/honeys'),
  getFeatured: () => api.get<Honey[]>('/honeys/featured'),
  getBySlug: (slug: string) => api.get<Honey>(`/honeys/${slug}`),
};
```
